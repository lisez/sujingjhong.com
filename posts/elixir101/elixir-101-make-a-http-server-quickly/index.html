function serverHandler(req, res){
  switch (req.url){
    case '/':{
      res.writeHead(200, {'Content-Type':'text/html'});
      res.write('Hello world');
      res.end();
      break;
    }
      
    default:{
      res.writeHead(404, {'Content-Type':'text/html'});

      break;
    }
  }
}

const server = http.createServer(serverHandler);
